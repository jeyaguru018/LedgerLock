        const pwdInput = document.getElementById('password');
        const strengthText = document.getElementById('pwd-strength-text');
        const strengthBars = document.querySelectorAll('.flex.gap-xs.h-1 > div');

        pwdInput.addEventListener('input', (e) => {
            const val = e.target.value;
            let strength = 0;
            
            if (val.length > 5) strength++;
            if (val.length > 8 && /[A-Z]/.test(val) && /[0-9]/.test(val)) strength++;
            if (val.length > 12 && /[^A-Za-z0-9]/.test(val)) strength++;

            strengthBars.forEach((bar, index) => {
                bar.className = 'transition-all duration-300 w-1/3 ';
                if (index < strength) {
                    if (strength === 1) bar.classList.add('bg-error');
                    else if (strength === 2) bar.classList.add('bg-secondary');
                    else bar.classList.add('bg-primary');
                } else {
                    bar.classList.add('bg-surface');
                }
            });

            if (strength === 0) strengthText.textContent = '';
            else if (strength === 1) { strengthText.textContent = 'Weak'; strengthText.className = 'font-label-md text-label-md text-error'; }
            else if (strength === 2) { strengthText.textContent = 'Medium'; strengthText.className = 'font-label-md text-label-md text-secondary'; }
            else { strengthText.textContent = 'Strong'; strengthText.className = 'font-label-md text-label-md text-primary'; }
        });

        // Micro-interactions and simulation logic
        document.addEventListener('DOMContentLoaded', () => {
            const slider = document.getElementById('concurrentSlider');
            const sliderValDisplay = document.getElementById('sliderValue');
            const runBtn = document.getElementById('runTestBtn');
            const sentCounter = document.getElementById('reqSentCounter');
            const appliedCounter = document.getElementById('reqAppliedCounter');
            const tpsCounter = document.getElementById('tpsCounter');
            const logContainer = document.getElementById('logContainer');
            const emptyState = document.getElementById('emptyLogState');
            
            let isRunning = false;

            slider.addEventListener('input', (e) => {
                sliderValDisplay.textContent = e.target.value;
            });

            function generateLogEntry(id, isSuccess) {
                const now = new Date();
                const timeStr = `${now.getHours().toString().padStart(2,'0')}:${now.getMinutes().toString().padStart(2,'0')}:${now.getSeconds().toString().padStart(2,'0')}.${now.getMilliseconds().toString().padStart(3,'0')}`;
                const threadId = `0x${Math.floor(Math.random()*16777215).toString(16).padStart(6,'0')}`;
                const amount = (Math.random() * 500).toFixed(2);
                
                const entry = document.createElement('div');
                entry.className = 'grid grid-cols-12 gap-sm px-sm py-xs border-b border-[#1F2023] log-entry-enter hover:bg-[#1A1C20] transition-colors';
                
                const statusHtml = isSuccess 
                    ? `<div class="col-span-2 text-right text-primary flex items-center justify-end gap-xs"><span class="w-1.5 h-1.5 rounded-full bg-primary"></span> OK</div>`
                    : `<div class="col-span-2 text-right text-error flex items-center justify-end gap-xs"><span class="w-1.5 h-1.5 rounded-full bg-error"></span> REJECT</div>`;

                entry.innerHTML = `
                    <div class="col-span-2 text-on-surface-variant">${timeStr}</div>
                    <div class="col-span-3 text-[#c0c7d3]">TH-${threadId.toUpperCase()}</div>
                    <div class="col-span-2 text-on-surface">DEPOSIT</div>
                    <div class="col-span-3 text-right text-on-surface">+$${amount}</div>
                    ${statusHtml}
                `;
                return entry;
            }

            runBtn.addEventListener('click', () => {
                if(isRunning) return;
                isRunning = true;
                
                // Reset UI
                sentCounter.textContent = '0';
                appliedCounter.textContent = '0';
                tpsCounter.textContent = '0';
                if(emptyState) emptyState.style.display = 'none';
                logContainer.innerHTML = '';
                
                const targetRequests = parseInt(slider.value, 10);
                let currentReq = 0;
                let appliedReq = 0;
                
                runBtn.innerHTML = `<span class="material-symbols-outlined animate-spin">sync</span> Running...`;
                runBtn.classList.remove('bg-primary-container', 'text-on-primary-fixed');
                runBtn.classList.add('bg-surface-variant', 'text-on-surface-variant', 'cursor-not-allowed');
                
                const startTime = Date.now();

                const interval = setInterval(() => {
                    if (currentReq >= targetRequests) {
                        clearInterval(interval);
                        isRunning = false;
                        runBtn.innerHTML = `<span class="material-symbols-outlined">rocket_launch</span> Run Concurrency Test`;
                        runBtn.classList.add('bg-primary-container', 'text-on-primary-fixed');
                        runBtn.classList.remove('bg-surface-variant', 'text-on-surface-variant', 'cursor-not-allowed');
                        return;
                    }

                    // Batch sizes to simulate concurrency spikes
                    const batchSize = Math.min(Math.floor(Math.random() * (targetRequests / 10) + 1), targetRequests - currentReq);
                    
                    for(let i=0; i<batchSize; i++) {
                        currentReq++;
                        // Simulate occasional optimistic locking failures (very rare)
                        const isSuccess = Math.random() > 0.02; 
                        if(isSuccess) appliedReq++;

                        // Add log (keep max 50 in DOM for performance)
                        const logEntry = generateLogEntry(currentReq, isSuccess);
                        logContainer.prepend(logEntry);
                        if(logContainer.children.length > 50) {
                            logContainer.removeChild(logContainer.lastChild);
                        }
                    }

                    // Update Counters
                    sentCounter.textContent = currentReq;
                    appliedCounter.textContent = appliedReq;
                    
                    // TPS calculation
                    const elapsedSec = (Date.now() - startTime) / 1000;
                    if(elapsedSec > 0) {
                        tpsCounter.textContent = Math.floor(currentReq / elapsedSec);
                    }

                }, 50); // Fast interval for visual effect
            });
        });
